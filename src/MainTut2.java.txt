public class Main{
    // --------Classes----------------
    // public :- access this Classes anywhere in the application
    // Main :- its class name, should be in the pascalCase.
    // Note :- Main class is the entry point of the application
    public static void main(String[] args){
        // --------Function(methods)-----------------
        // public :- access this function(method) from anywhere in the application
        // static :- we can directly use this function(method), no need to create object from respective class.
        // void :- its indicates that function(method) does not return anythings(values).
        // main :- its function(method) name, should be in the camelCase.
        int number=8;
        int number2=8;
        System.out.println("Described Java file's basic structure");
        System.out.println(number + number2);
        // To view/print the values
    }
}
